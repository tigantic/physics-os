## 2. DOMAIN NORMALIZATION

The key insight: every physics domain has a "complexity knob" — a parameter that makes the physics harder. To compare across domains, we normalize these to a universal scale.

| Domain Class | Complexity Parameter | Normalized Range |
|--------------|---------------------|------------------|
| Fluids | Reynolds number (Re) | 10⁰ → 10⁷ |
| Quantum | Coupling strength (g) | 0 → ∞ |
| Thermal | Rayleigh number (Ra) | 10³ → 10¹² |

| Plasma | Lundquist number (S) | 10² → 10⁸ |
| Materials | Strain / yield ratio | 0 → 10 |
| Quantum Many-Body | System size × interaction | 0 → ∞ |

**Normalized complexity ξ:**

```
ξ = log₁₀(parameter / parameter_trivial) / log₁₀(parameter_hard / parameter_trivial)
```

Maps every domain to ξ ∈ [0, 1] where:
- ξ = 0: trivial limit (rank should be minimal)
- ξ = 1: extreme limit (if rank explodes, it happens here)

## 5. THE MEASUREMENT FUNCTION

```python
def measure_domain(
    domain: PhysicsDomain,
    n_bits: int,
    complexity_value: float,
    reference_dense: bool = True
) -> RankMeasurement:
    """
    Single measurement point for one domain at one complexity level.
    """
    
    # 1. Configure the domain
    config = domain.configure(
        n_bits=n_bits,
        **{domain.complexity_param: complexity_value}
    )
    
    # 2. Generate or solve the physics
    if domain.is_static:
        # Static field (e.g., Poisson solution)
        qtt_state = domain.solve(config)
    else:
        # Time-evolving (e.g., Navier-Stokes)
        qtt_state = domain.evolve_to_steady(config)
    
    # 3. Extract QTT cores
    cores = qtt_state.cores
    
    # 4. Compute all metrics
    bond_dims = [cores[k].shape[2] for k in range(len(cores) - 1)]
    
    sv_spectra = []
    sv_decay_exponents = []
    bond_entropies = []
    sv_gap_ratios = []
    
    for k in range(len(cores) - 1):
        # SVD at this bond
        U, S, Vh = torch.linalg.svd(
            cores[k].reshape(-1, cores[k].shape[2]), 
            full_matrices=False
        )
        
        # Full spectrum
        sv = S.cpu().numpy().tolist()
        sv_spectra.append(sv)
        
        # Decay exponent: fit log(σ) vs log(i)
        if len(sv) > 2:
            log_i = np.log(np.arange(1, len(sv) + 1))
            log_sv = np.log(np.array(sv) + 1e-16)
            alpha, _ = np.polyfit(log_i, log_sv, 1)
            sv_decay_exponents.append(-alpha)  # Positive = decay
        else:
            sv_decay_exponents.append(0.0)
        
        # Entropy
        p = (S ** 2) / (S ** 2).sum()
        entropy = -(p * torch.log(p + 1e-16)).sum().item()
        bond_entropies.append(entropy)
        
        # Spectral gap
        if len(sv) > 1:
            sv_gap_ratios.append(sv[0] / (sv[1] + 1e-16))
        else:
            sv_gap_ratios.append(float('inf'))
    
    # 5. Profile shape analysis
    bd = np.array(bond_dims)
    gradient = np.diff(bd).tolist() if len(bd) > 1 else [0.0]
    curvature = np.diff(bd, n=2).tolist() if len(bd) > 2 else [0.0]
    smoothness = np.linalg.norm(gradient)
    
    # Symmetry: correlation of left half with reversed right half
    mid = len(bd) // 2
    if mid > 0:
        left = bd[:mid]
        right = bd[-mid:][::-1]
        symmetry = np.corrcoef(left, right)[0, 1] if len(left) > 1 else 1.0
    else:
        symmetry = 1.0
    
    # 6. Reference comparison (if requested)
    if reference_dense and n_bits <= 9:  # Only feasible up to 512³
        dense = qtt_state.to_dense()
        reconstructed = qtt_state.reconstruct()
        error = torch.norm(dense - reconstructed) / torch.norm(dense)
        reconstruction_error = error.item()
    else:
        reconstruction_error = None
    
    # 7. Storage
    dense_bytes = (2 ** (3 * n_bits)) * 4 * 3  # 3 components, float32
    qtt_bytes = sum(c.numel() * 4 for c in cores)
    
    # 8. Assemble record
    return RankMeasurement(
        domain_id=domain.taxonomy_id,
        domain_name=domain.name,
        pde_class=domain.pde_class,
        n_bits=n_bits,
        n_cells=2 ** (3 * n_bits),
        complexity_param_name=domain.complexity_param,
        complexity_param_value=complexity_value,
        complexity_normalized=domain.normalize_complexity(complexity_value),
        n_sites=3 * n_bits,  # 3D QTT
        bond_dimensions=bond_dims,
        sv_spectra=sv_spectra,
        max_rank=max(bond_dims),
        mean_rank=float(np.mean(bond_dims)),
        median_rank=int(np.median(bond_dims)),
        rank_std=float(np.std(bond_dims)),
        peak_rank_site=int(np.argmax(bond_dims)),
        rank_utilization=float(np.mean(bond_dims)) / max(bond_dims),
        profile_gradient=gradient,
        profile_curvature=curvature,
        profile_smoothness=smoothness,
        profile_symmetry=float(symmetry) if not np.isnan(symmetry) else 0.0,
        sv_decay_exponents=sv_decay_exponents,
        sv_decay_mean=float(np.mean(sv_decay_exponents)),
        sv_gap_ratios=sv_gap_ratios,
        bond_entropies=bond_entropies,
        total_entropy=sum(bond_entropies),
        entropy_density=sum(bond_entropies) / len(bond_entropies),
        dense_bytes=dense_bytes,
        qtt_bytes=qtt_bytes,
        compression_ratio=dense_bytes / qtt_bytes,
        reconstruction_error=reconstruction_error,
        energy_conservation=qtt_state.energy_drift if hasattr(qtt_state, 'energy_drift') else None,
        physics_invariant_passed=qtt_state.invariants_ok if hasattr(qtt_state, 'invariants_ok') else True,
        wall_time_s=qtt_state.wall_time,
        gpu_mem_mb=qtt_state.peak_memory_mb,
        timestamp=datetime.utcnow().isoformat(),
        config_hash=hashlib.sha256(str(config).encode()).hexdigest()[:16],
        result_hash=hashlib.sha256(torch.cat([c.flatten() for c in cores]).numpy().tobytes()).hexdigest()[:16]
    )
```

## 6. THE CAMPAIGN RUNNER

```python
def run_rank_atlas_campaign(
    domains: List[PhysicsDomain],
    output_path: str = "rank_atlas.parquet",
    n_bits_list: List[int] = [6, 7, 8, 9],
    n_complexity_points: int = 10,
    resume: bool = True
) -> pd.DataFrame:
    """
    Full measurement campaign across all domains.
    """
    
    measurements = []
    
    # Resume from existing if requested
    if resume and os.path.exists(output_path):
        existing = pd.read_parquet(output_path)
        completed = set(zip(existing['domain_id'], existing['n_bits'], existing['complexity_normalized']))
        measurements = existing.to_dict('records')
        print(f"Resuming from {len(measurements)} existing measurements")
    else:
        completed = set()
    
    total = len(domains) * len(n_bits_list) * n_complexity_points
    pbar = tqdm(total=total, desc="Rank Atlas")
    
    for domain in domains:
        # Complexity sweep for this domain
        xi_values = np.linspace(0, 1, n_complexity_points)
        complexity_values = [domain.denormalize_complexity(xi) for xi in xi_values]
        
        for n_bits in n_bits_list:
            for xi, complexity in zip(xi_values, complexity_values):
                
                # Skip if already done
                key = (domain.taxonomy_id, n_bits, round(xi, 2))
                if key in completed:
                    pbar.update(1)
                    continue
                
                try:
                    measurement = measure_domain(
                        domain=domain,
                        n_bits=n_bits,
                        complexity_value=complexity,
                        reference_dense=(n_bits <= 8)  # Only check error for smaller grids
                    )
                    measurements.append(asdict(measurement))
                    
                except Exception as e:
                    print(f"FAILED: {domain.taxonomy_id} @ n_bits={n_bits}, ξ={xi:.2f}: {e}")
                    # Record failure
                    measurements.append({
                        'domain_id': domain.taxonomy_id,
                        'n_bits': n_bits,
                        'complexity_normalized': xi,
                        'failed': True,
                        'error': str(e)
                    })
                
                pbar.update(1)
                
                # Checkpoint every 100 measurements
                if len(measurements) % 100 == 0:
                    df = pd.DataFrame(measurements)
                    df.to_parquet(output_path)
    
    pbar.close()
    
    # Final save
    df = pd.DataFrame(measurements)
    df.to_parquet(output_path)
    
    return df
```

## 7. ANALYSIS PIPELINE

```python
def analyze_rank_atlas(df: pd.DataFrame) -> dict:
    """
    Core analysis: find universality classes.
    """
    
    results = {}
    
    # =========================================
    # 1. SCALING CLASS CLASSIFICATION
    # =========================================
    
    def classify_scaling(group):
        """
        Fit rank vs complexity and classify.
        """
        xi = group['complexity_normalized'].values
        rank = group['max_rank'].values
        
        if len(xi) < 3:
            return 'insufficient_data'
        
        # Fit: rank = a * xi^b + c
        # Log-transform: log(rank - c) = log(a) + b * log(xi)
        
        # Try constant fit (Class A)
        rank_var = np.var(rank)
        rank_mean = np.mean(rank)
        cv = np.sqrt(rank_var) / rank_mean if rank_mean > 0 else float('inf')
        
        if cv < 0.1:  # Less than 10% variation
            return 'A'  # Rank-independent
        
        # Try log fit (Class B)
        try:
            log_xi = np.log(xi + 0.01)
            slope_log, intercept_log, r_log, _, _ = linregress(log_xi, rank)
            
            # Try power-law fit (Class C)
            log_rank = np.log(rank + 1)
            slope_pow, intercept_pow, r_pow, _, _ = linregress(np.log(xi + 0.01), log_rank)
            
            # Decide based on R²
            r2_log = r_log ** 2
            r2_pow = r_pow ** 2
            
            if r2_log > 0.8 and slope_log > 0:
                if slope_pow < 0.3:
                    return 'B'  # Logarithmic
                else:
                    return f'C:{slope_pow:.2f}'  # Power-law with exponent
            
            if r2_pow > 0.8:
                if slope_pow > 1.5:
                    return 'D'  # Exponential-ish (dangerous)
                else:
                    return f'C:{slope_pow:.2f}'
            
            return 'B'  # Default to well-behaved
            
        except:
            return 'unknown'
    
    # Classify each domain
    scaling_classes = df.groupby('domain_id').apply(classify_scaling)
    results['scaling_classes'] = scaling_classes.to_dict()
    
    # =========================================
    # 2. UNIVERSALITY CLUSTERING
    # =========================================
    
    # Feature vector per domain: [mean_rank, rank_std, sv_decay_mean, entropy_density, profile_symmetry]
    domain_features = df.groupby('domain_id').agg({
        'max_rank': 'mean',
        'rank_std': 'mean',
        'sv_decay_mean': 'mean',
        'entropy_density': 'mean',
        'profile_symmetry': 'mean',
        'compression_ratio': 'mean'
    }).dropna()
    
    # Normalize features
    from sklearn.preprocessing import StandardScaler
    from sklearn.cluster import KMeans, DBSCAN
    from sklearn.metrics import silhouette_score
    
    X = StandardScaler().fit_transform(domain_features.values)
    
    # Try different numbers of clusters
    silhouettes = []
    for k in range(2, 8):
        km = KMeans(n_clusters=k, random_state=42)
        labels = km.fit_predict(X)
        sil = silhouette_score(X, labels)
        silhouettes.append((k, sil))
    
    best_k = max(silhouettes, key=lambda x: x[1])[0]
    
    km = KMeans(n_clusters=best_k, random_state=42)
    cluster_labels = km.fit_predict(X)
    
    results['optimal_clusters'] = best_k
    results['silhouette_scores'] = silhouettes
    results['cluster_assignments'] = dict(zip(domain_features.index, cluster_labels))
    
    # =========================================
    # 3. PDE CLASS CORRELATION
    # =========================================
    
    # Does cluster assignment correlate with PDE type?
    df_with_clusters = df.merge(
        pd.DataFrame({'domain_id': domain_features.index, 'cluster': cluster_labels}),
        on='domain_id'
    )
    
    pde_cluster_table = pd.crosstab(df_with_clusters['pde_class'], df_with_clusters['cluster'])
    results['pde_cluster_correlation'] = pde_cluster_table.to_dict()
    
    # =========================================
    # 4. GRID INDEPENDENCE TEST
    # =========================================
    
    # For each domain: does max_rank change with n_bits at fixed complexity?
    def test_grid_independence(group):
        if group['n_bits'].nunique() < 3:
            return None
        pivot = group.pivot_table(values='max_rank', index='complexity_normalized', columns='n_bits')
        # Compute variance across grid sizes at each complexity
        variance_ratio = pivot.std(axis=1).mean() / pivot.mean(axis=1).mean()
        return variance_ratio
    
    grid_independence = df.groupby('domain_id').apply(test_grid_independence)
    results['grid_independence'] = grid_independence.dropna().to_dict()
    
    # Domains where rank varies >10% with grid = NOT grid-independent
    grid_dependent = [d for d, v in results['grid_independence'].items() if v > 0.1]
    results['grid_dependent_domains'] = grid_dependent
    
    # =========================================
    # 5. THE BIG QUESTION: ONE REGIME?
    # =========================================
    
    # If all domains are Class A AND cluster into 1 group...
    all_class_a = all(c == 'A' for c in results['scaling_classes'].values())
    one_cluster = results['optimal_clusters'] == 1 or (
        results['optimal_clusters'] == 2 and 
        min(silhouettes, key=lambda x: abs(x[0] - 1))[1] > 0.5
    )
    
    results['one_regime_hypothesis'] = {
        'all_class_a': all_class_a,
        'single_cluster': one_cluster,
        'verdict': 'CONFIRMED' if (all_class_a and one_cluster) else 'REJECTED' if not all_class_a else 'PARTIAL'
    }
    
    return results
```

```python
    for domain_id in sorted(domains):
        domain_df = df[df['domain_id'] == domain_id]
        max_bits = domain_df['n_bits'].max()
        peak = domain_df[(domain_df['n_bits'] == max_bits) & 
                        (domain_df['complexity_normalized'] > 0.9)]
        
        if len(peak) > 0:
            profile = peak.iloc[0]['bond_dimensions']
            if profile and len(profile) > 0:
                # Normalize to 27 sites (pad or truncate)
                profile = np.array(profile)
                if len(profile) < 27:
                    profile = np.pad(profile, (0, 27 - len(profile)))
                else:
                    profile = profile[:27]
                peak_profiles.append(profile)
                domain_labels.append(domain_id)
    
    if peak_profiles:
        profile_matrix = np.array(peak_profiles)
        
        fig, ax = plt.subplots(figsize=(14, len(domain_labels) * 0.3 + 2))
        im = ax.imshow(profile_matrix, aspect='auto', cmap='viridis')
        
        ax.set_yticks(range(len(domain_labels)))
        ax.set_yticklabels(domain_labels, fontsize=6)
        ax.set_xlabel('QTT Site', fontsize=10)
        ax.set_title('Rank Profile Heatmap (ξ=1)', fontsize=12)
        
        plt.colorbar(im, ax=ax, label='Bond Dimension')
        plt.tight_layout()
        plt.savefig(f'{output_dir}/profile_heatmap.png', dpi=150)
        plt.close()
    
    # =========================================
    # PLOT 4: SCALING CLASS DISTRIBUTION
    # =========================================
    
    class_counts = pd.Series(results['scaling_classes']).value_counts()
    
    fig, ax = plt.subplots(figsize=(8, 6))
    
    colors_map = {'A': 'green', 'B': 'blue', 'D': 'red'}
    bar_colors = [colors_map.get(c[0], 'orange') for c in class_counts.index]
    
    ax.bar(range(len(class_counts)), class_counts.values, color=bar_colors)
    ax.set_xticks(range(len(class_counts)))
    ax.set_xticklabels(class_counts.index, rotation=45)
    ax.set_ylabel('Number of Domains')
    ax.set_title('Scaling Class Distribution')
    
    # Add verdict
    verdict = results['one_regime_hypothesis']['verdict']
    ax.text(0.95, 0.95, f"One Regime: {verdict}", 
           transform=ax.transAxes, ha='right', va='top',
           fontsize=12, fontweight='bold',
           color='green' if verdict == 'CONFIRMED' else 'red')
    
    plt.tight_layout()
    plt.savefig(f'{output_dir}/scaling_classes.png', dpi=150)
    plt.close()
    
    # =========================================
    # SUMMARY REPORT
    # =========================================
    
    with open(f'{output_dir}/ATLAS_SUMMARY.md', 'w') as f:
        f.write("# QTT RANK ATLAS — SUMMARY\n\n")
        f.write(f"**Total Measurements:** {len(df)}\n")
        f.write(f"**Domains Measured:** {df['domain_id'].nunique()}\n")
        f.write(f"**Grid Sizes:** {sorted(df['n_bits'].unique())}\n\n")
        
        f.write("## Scaling Classes\n\n")
        for cls, count in class_counts.items():
            f.write(f"- **{cls}**: {count} domains\n")
        
        f.write(f"\n## Universality Clusters\n\n")
        f.write(f"**Optimal K:** {results['optimal_clusters']}\n")
        f.write(f"**Silhouette Scores:** {results['silhouette_scores']}\n\n")
        
        f.write("## One Regime Hypothesis\n\n")
        f.write(f"- All Class A: {results['one_regime_hypothesis']['all_class_a']}\n")
        f.write(f"- Single Cluster: {results['one_regime_hypothesis']['single_cluster']}\n")
        f.write(f"- **VERDICT: {results['one_regime_hypothesis']['verdict']}**\n")
```
